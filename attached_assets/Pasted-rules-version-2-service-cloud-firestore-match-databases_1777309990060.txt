rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function userEmail() {
      return request.auth.token.email;
    }

    function isSelf(userId) {
      return signedIn() && request.auth.uid == userId;
    }

    function membershipId(orgId, role, email) {
      return "mbr_" + orgId + "_" + role + "_" + email;
    }

    function hasActiveMembership(orgId) {
      return signedIn()
        && exists(/databases/$(database)/documents/memberships/$(request.auth.uid + "_" + orgId));
    }

    function isAdminOfOrg(orgId) {
      return signedIn()
        && exists(/databases/$(database)/documents/memberships/$(request.auth.uid + "_" + orgId))
        && get(/databases/$(database)/documents/memberships/$(request.auth.uid + "_" + orgId)).data.role == "admin"
        && get(/databases/$(database)/documents/memberships/$(request.auth.uid + "_" + orgId)).data.status == "active";
    }

    match /users/{userId} {
      allow read: if isSelf(userId);

      allow create: if isSelf(userId)
        && request.resource.data.uid == request.auth.uid;

      allow update: if isSelf(userId)
        && resource.data.uid == request.auth.uid
        && request.resource.data.uid == request.auth.uid;

      allow delete: if false;
    }

    match /userIndex/{readableUserId} {
      allow read: if signedIn()
        && request.auth.uid == resource.data.uid;

      allow create: if signedIn()
        && request.auth.uid == request.resource.data.uid;

      allow update: if signedIn()
        && request.auth.uid == resource.data.uid
        && request.auth.uid == request.resource.data.uid;

      allow delete: if false;
    }

    match /organizations/{organizationId} {
      allow read: if signedIn();

      allow create: if signedIn()
        && request.resource.data.createdBy == request.auth.uid
        && request.resource.data.createdByEmail == userEmail()
        && request.resource.data.source == "create_org_form"
        && request.resource.data.schemaVersion == 2;

      allow update: if isAdminOfOrg(organizationId);

      allow delete: if isAdminOfOrg(organizationId);
    }

    match /memberships/{membershipId} {
      allow read: if signedIn()
        && (
          request.auth.uid == resource.data.userId
          || isAdminOfOrg(resource.data.organizationId)
        );

      // First admin membership created during organization creation
      allow create: if signedIn()
        && request.resource.data.userId == request.auth.uid
        && request.resource.data.email == userEmail()
        && request.resource.data.role == "admin"
        && request.resource.data.status == "active"
        && request.resource.data.source == "create_org_form"
        && request.resource.data.schemaVersion == 2;

      allow update: if false;
      allow delete: if false;
    }

    match /invitations/{inviteId} {
      allow read: if signedIn()
        && (
          request.auth.token.email == resource.data.email
          || request.auth.token.email == resource.data.invitedEmail
          || isAdminOfOrg(resource.data.organizationId)
        );

      allow create: if signedIn()
        && isAdminOfOrg(request.resource.data.organizationId)
        && request.resource.data.invitedBy == request.auth.uid
        && request.resource.data.status == "pending"
        && request.resource.data.schemaVersion == 2;

      allow update: if signedIn()
        && (
          request.auth.token.email == resource.data.email
          || request.auth.token.email == resource.data.invitedEmail
        );

      allow delete: if false;
    }

    match /tasks/{taskId} {
      allow read: if signedIn()
        && request.auth.uid == resource.data.userId;

      allow create: if signedIn()
        && request.resource.data.userId == request.auth.uid;

      allow update: if signedIn()
        && resource.data.userId == request.auth.uid
        && request.resource.data.userId == request.auth.uid;

      allow delete: if signedIn()
        && resource.data.userId == request.auth.uid;
    }

    match /categories/{categoryId} {
      allow read: if signedIn()
        && request.auth.uid == resource.data.userId;

      allow create: if signedIn()
        && request.resource.data.userId == request.auth.uid;

      allow update: if signedIn()
        && resource.data.userId == request.auth.uid
        && request.resource.data.userId == request.auth.uid;

      allow delete: if signedIn()
        && resource.data.userId == request.auth.uid;
    }
  }
}
